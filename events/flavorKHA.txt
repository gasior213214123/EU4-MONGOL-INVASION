# Mongol Invasion

	title = "flavor_kha.EVTNAME1"
	desc = "flavor_kha.EVTDESC1"

	trigger = {
		tag = KHA
	}

	}
	
	option = {

        change_unit_type = high_american

        }
        inherit = OIR
        #MNG = {
    	#	set_mandate = 0
        #}
        set_government_rank = 3

			add_opinion = { who = ROOT modifier = opinion_invasion }
		}
	}
}
